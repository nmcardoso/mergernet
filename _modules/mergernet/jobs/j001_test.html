def run():
  NAME = 'j001_test'

  ds = Dataset(config=Dataset.registry.BIN_SDSS_128)
  model = HyperModel(dataset=ds, name=NAME, epochs=10)
  model.hypertrain(
    n_trials=1,
    hyperparameters=HyperParameterSet(hp),
    pruner='hyperband',
    objective_metric='val_loss',
    objective_direction='minimize',
    resume=False,
    save_model=True
  )
